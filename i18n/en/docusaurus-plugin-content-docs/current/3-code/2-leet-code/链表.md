### Reverse a linked list by hand

```js
class ListNode {
    constructor(value) {
        this.value = value;
        this.next = null;
    }
}

function reverseLinkedList(head) {
    let prev = null;
    let current = head;
    let next = null;

    while (current !== null) {
        // Save next node
        next = current.next;
        // Reverse current node's pointer
        current.next = prev;
        // Advance prev and current
        prev = current;
        current = next;
    }

    // prev now points to the new head
    return prev;
}

// Build the list
const head = new ListNode(1);
head.next = new ListNode(2);
head.next.next = new ListNode(3);
head.next.next.next = new ListNode(4);
head.next.next.next.next = new ListNode(5);

// Print original list
let current = head;
while (current !== null) {
    console.log(current.value);
    current = current.next;
}

// Reverse
const newHead = reverseLinkedList(head);

// Print reversed list
current = newHead;
while (current !== null) {
    console.log(current.value);
    current = current.next;
}
```

### Reverse Nodes in k-Group

```javascript
function reverseKGroup(head, k) {
    let count = 0;
    let node = head;
    while (node && count < k) {
        node = node.next;
        count++;
    }
    if (count === k) {
        node = reverseKGroup(node, k);
        while (count > 0) {
            const temp = head.next;
            head.next = node;
            node = head;
            head = temp;
            count--;
        }
        head = node;
    }
    return head;
}
```
