### Add Binary

```js
var addBinary = function(a, b) {
    let i = a.length - 1       // max index of a
    let j = b.length - 1       // max index of b
    let res = ''
    let c = 0                        // carry
    while( i >= 0 || j >=0 || c ){   // add from right to left
      let aa = +a[ i-- ] || 0
      let bb = +b[ j-- ] || 0
      let total = aa + bb + c
      c = total / 2 | 0
      total %= 2
      res = total + res;
    }
    return res
};
```

**Example 1:**

```
Input: nums1 = [1,2,3,0,0,0], m = 3, nums2 = [2,5,6], n = 3
Output: [1,2,2,3,5,6]
Explanation: Merge [1,2,3] and [2,5,6].
The merged result is [1,2,2,3,5,6]; the bold italic items come from nums1.
```

**Example 2:**

```
Input: nums1 = [1], m = 1, nums2 = [], n = 0
Output: [1]
Explanation: Merge [1] and [].
Result is [1].
```

**Example 3:**

```
Input: nums1 = [0], m = 0, nums2 = [1], n = 1
Output: [1]
Explanation: Merge [] and [1].
Result is [1].
Note: since m = 0, nums1 has no elements; the trailing 0 is just to ensure the merge can fit in nums1.
```

```ts
function merge(nums1: number[], m: number, nums2: number[], n: number): void {
    // From the back, two pointers, total a+b -1 iterations
    for(let i = m - 1, j = n - 1, k = n + m - 1; j >= 0; --k) {
        nums1[k] = i >= 0 && nums1[i] > nums2[j] ? nums1[i--] : nums2[j--]
    }

};
```

## [125. Valid Palindrome](https://leetcode.cn/problems/valid-palindrome/)

**Example 1:**

```
Input: s = "A man, a plan, a canal: Panama"
Output: true
Explanation: "amanaplanacanalpanama" is a palindrome.
```

**Example 2:**

```
Input: s = "race a car"
Output: false
Explanation: "raceacar" is not a palindrome.
```

**Example 3:**

```
Input: s = " "
Output: true
Explanation: After removing non-alphanumeric characters, s is an empty string "".
An empty string reads the same forwards and backwards, so it's a palindrome.
```

```js
function isPalindrome(s: string): boolean {
    s = s.toLocaleLowerCase().replace(/([^a-zA-Z0-9])/g, '')
    let left = 0, right = s.length - 1;
    while(right >= left){
        if(s[left] !== s[right]) return false;
        right--
        left++
    }
    return true;
};
```

 ```js
function isSubsequence(s: string, t: string): boolean {
    let l = 0;
    for(let char of t){
        if(s[l] === char) l++;
    }
    return l === s.length;
};
 ```

## [167. Two Sum II - Input Array Is Sorted](https://leetcode.cn/problems/two-sum-ii-input-array-is-sorted/)

**Example 1:**

```
Input: numbers = [2,7,11,15], target = 9
Output: [1,2]
Explanation: 2 + 7 = 9. Thus index1 = 1, index2 = 2. Return [1, 2].
```

**Example 2:**

```
Input: numbers = [2,3,4], target = 6
Output: [1,3]
Explanation: 2 + 4 = 6. Thus index1 = 1, index2 = 3. Return [1, 3].
```

**Example 3:**

```
Input: numbers = [-1,0], target = -1
Output: [1,2]
Explanation: -1 + 0 = -1. Thus index1 = 1, index2 = 2. Return [1, 2].
```

```js
function twoSum(numbers: number[], target: number): number[] {
    let [i, j] = [0, numbers.length - 1];
    while(i < j){
        let sum = numbers[i] + numbers[j];
        if(sum < target) i++
        else if(sum > target) j--
        else return [i + 1, j + 1]
    }
};
```

```js
function maxArea(height: number[]): number {
    let [i, j, area] = [0, height.length - 1, 0];
    while(i < j){
        const newArea = Math.min(height[i], height[j]) * (j - i);
        if(newArea > area) area = newArea;
        if(height[i] < height[j]) i++;
        else j--
    }
    return area;
};
```

## [15. 3Sum](https://leetcode.cn/problems/3sum/)

**Example 1:**

```
Input: nums = [-1,0,1,2,-1,-4]
Output: [[-1,-1,2],[-1,0,1]]
Explanation:
nums[0] + nums[1] + nums[2] = (-1) + 0 + 1 = 0.
nums[1] + nums[2] + nums[4] = 0 + 1 + (-1) = 0.
nums[0] + nums[3] + nums[4] = (-1) + 2 + (-1) = 0.
The unique triplets are [-1,0,1] and [-1,-1,2].
The order of output and triplets does not matter.
```

**Example 2:**

```
Input: nums = [0,1,1]
Output: []
Explanation: The only possible triplet does not sum to 0.
```

**Example 3:**

```
Input: nums = [0,0,0]
Output: [[0,0,0]]
Explanation: The only possible triplet sums to 0.
```

```js
function threeSum(nums: number[]): number[][] {
    nums.sort((a, b) => a - b);
    const len = nums.length;
    const ans = [];
    for (let i = 0; i < len; i++) {
        if (i > 0 && nums[i] === nums[i - 1]) continue;
        let j = i + 1, k = len - 1;
        while (j < k) {
            while (j > i + 1 && j < len && nums[j] === nums[j - 1]) j++;
            if (j >= k) break;
            const sum = nums[i] + nums[j] + nums[k];
            if (sum === 0) {
                ans.push([nums[i], nums[j], nums[k]]);
                j++;
            } else if (sum > 0) {
                k--;
            } else {
                j++
            }
        }
    }

    return ans;

};
```
