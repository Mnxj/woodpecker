### Valid Parentheses

```js
// Stack + map; check the previous open bracket: 1. matches → pop; 2. opens → push
function isValid(s) {
    const stack = [];
    const leftBrackets = ['(', '{', '['];
    const rightBrackets = [')', '}', ']'];

    for (let i = 0; i < s.length; i++) {
      const char = s[i];

      if (leftBrackets.includes(char)) {
        stack.push(char);
      } else if (rightBrackets.includes(char)) {
        if (stack.length === 0 || leftBrackets[rightBrackets.indexOf(char)] !== stack.pop()) {
          return false;
        }
      }
    }

    return stack.length === 0;
  }

  // Tests
  console.log(isValid('()'));
  console.log(isValid('()[]{}'));
  console.log(isValid('(]'));
  console.log(isValid('([)]'));
  console.log(isValid('{[]}'));
```

### Merge unsorted intervals

```js
// Implement merging of unsorted intervals
/**
    Input  [[1,4], [3,8], [10,15], [11,18], [20,27], [14,15], [19,28]]
    Expected output  [[1, 8], [10,18], [19,28]]
*/
function f(nums) {
    nums.sort((a, b) => a[0] - b[0]);
    const ans = [];
    for(let p of nums) {
        const m = ans.length;
        if(m && p[0] <= ans[m - 1][1]){
            ans[m - 1][1] = Math.max(ans[m - 1][1], p[1]);
        } else {
            ans.push(p);
        }
    }
    return ans;
}

const nums = [[1, 4], [3, 8], [10, 15], [11, 18], [20, 27], [14, 15], [19, 28]];
console.log(f(nums))
```

### Given two stacks (with `pop`, `push`, `getSize`), implement a queue with `dequeue` and `enqueue`

```js
class Queue {
    constructor() {
        this.inStack = [];
        this.outStack = [];
    }

    // Enqueue
    enqueue(value) {
        this.inStack.push(value);
    }

    // Dequeue
    dequeue() {
        if (this.outStack.length === 0) {
            // If outStack is empty, transfer all elements from inStack
            while (this.inStack.length > 0) {
                this.outStack.push(this.inStack.pop());
            }
        }
        // If outStack has elements, pop the top
        if (this.outStack.length > 0) {
            return this.outStack.pop();
        }
        // Both empty → queue is empty
        return null;
    }

    // Queue size
    getSize() {
        return this.inStack.length + this.outStack.length;
    }
}

// Usage
const queue = new Queue();
queue.enqueue(1);
queue.enqueue(2);
console.log(queue.dequeue()); // 1
queue.enqueue(3);
console.log(queue.dequeue()); // 2
console.log(queue.dequeue()); // 3
console.log(queue.getSize()); // 0
```

### Scenario: queues, event loop, chained calls

```js
// Implement LazyMan that can be called like:
// LazyMan("Hank") outputs:
// Hi! This is Hank!

// LazyMan("Hank").sleep(10).eat("dinner") outputs:
// Hi! This is Hank!
// // wait 10 seconds...
// Wake up after 10
// Eat dinner~

// LazyMan("Hank").eat("dinner").eat("supper") outputs:
// Hi This is Hank!
// Eat dinner~
// Eat supper~

// LazyMan("Hank").sleepFirst(5).eat("supper") outputs:
// // wait 5 seconds
// Wake up after 5
// Hi This is Hank!
// Eat supper
//

class LazyMan {
    constructor(name) {
      this.name = name;
      this.queue = [];
      setTimeout(() => {
        if(this.status !== 1){
            this.sayHi();
        }
        this.runQueue();
      }, 0);
    }

    sayHi() {
      console.log(`Hi! This is ${this.name}!`);
    }

    sleep(time) {
        this.queue.push(() => {
            setTimeout(() => {
              console.log(`Wake up after ${time}`);
              this.runQueue();
            }, time * 1000);
          });
      return this;
    }

    sleepFirst(time) {
      this.status = 1
      this.queue.unshift(() => {
        setTimeout(() => {
          console.log(`Wake up after ${time}`);
          this.sayHi();
          this.runQueue();
        }, time * 1000);
      });
      return this;
    }

    eat(food) {
        this.queue.push(() => {
            console.log(`Eat ${food}~`);
            this.runQueue();
          });
      return this;
    }

    runQueue() {
      const task = this.queue.shift();
      if (task) {
        task();
      }
    }
  }

  // Tests
  new LazyMan('Hank').sleep(10).eat('dinner');
//   new LazyMan('Hank').eat('dinner').eat('supper');
//   new LazyMan('Hank').sleepFirst(5).eat('supper');

```
