

### Implement a simple axios

```js
function simpleAxios(url, options = {}) {
    // Defaults
    const defaultOptions = {
        method: 'GET',
        headers: {
            'Content-Type': 'application/json'
        }
    };

    // Merge defaults with user options
    const mergedOptions = { ...defaultOptions, ...options };

    // Use fetch to make the request
    return fetch(url, mergedOptions)
        .then(response => {
            // Check status
            if (!response.ok) {
                throw new Error(`HTTP error! status: ${response.status}`);
            }
            // Parse response based on responseType
            if (mergedOptions.responseType === 'json') {
                return response.json();
            } else if (mergedOptions.responseType === 'text') {
                return response.text();
            } else {
                return response;
            }
        })
        .catch(error => {
            // Handle request errors
            console.error('Request failed:', error);
            throw error;
        });
}

// Usage
simpleAxios('https://api.example.com/data', {
    method: 'POST',
    headers: {
        'Authorization': 'Bearer your-token'
    },
    body: JSON.stringify({ key: 'value' }),
    responseType: 'json'
}).then(data => {
    console.log('Success:', data);
}).catch(error => {
    console.error('Error:', error);
});
```



### Benefits of wrapping axios

- Unified request configuration: set shared options for all API requests — headers, timeouts, base URL.

- Wrapping can centralize error handling — auto-handle network errors, timeouts, HTTP errors.
- You can add request and response interceptors for pre/post processing — auth tokens, response normalization.
- Code reuse: a single wrapped instance can be shared by many components/services, avoiding duplication.
- Easier maintenance and extension: centralized config simplifies future updates.
- Security: the wrapper can include security logic — auto CSRF token handling, CSRF prevention, etc.
- Easier testing: the wrapped axios instance is easier to unit-test and integration-test since request/response handling lives in an isolated module.
- Cleaner project structure — easier for others to understand and maintain.

A simple axios wrapper example:

```javascript
import axios from 'axios';

// Create an axios instance
const apiClient = axios.create({
    baseURL: 'https://api.example.com',
    timeout: 10000,
    headers: {
        'Content-Type': 'application/json',
    },
});

// Request interceptor
apiClient.interceptors.request.use(
    config => {
        // Do something before the request is sent
        return config;
    },
    error => {
        // Handle the request error
        return Promise.reject(error);
    }
);

// Response interceptor
apiClient.interceptors.response.use(
    response => {
        // Do something with the response data
        return response;
    },
    error => {
        // Handle the response error
        return Promise.reject(error);
    }
);

export default apiClient;
```


### Implementing simple hash routing

```js
const routes = {};

function registerRoute(path, handler) {
    routes[path] = handler;
}

function navigate(path) {
    if (routes[path]) {
        routes[path]();
    } else {
        console.log('Route not found');
    }
}

// Register routes
registerRoute('/page1', () => {
    console.log('This is page1');
});

registerRoute('/page2', () => {
    console.log('This is page2');
});

// Navigate
navigate('/page1');
navigate('/page2');
```

      let isToastShown = false;

### **How to handle massive concurrent requests on a page?**

- **Solutions**:
    - Use `Promise.all` for parallel requests.
    - Use a request queue to limit concurrency.
    - Use caching (e.g. Redis) to reduce duplicate requests.

### Write a function. Takes a pool of promises and a concurrency limit; returns results in the order of the pool.

```js
function promisePool(promises, concurrency) {
  return new Promise((resolve, reject) => {
    let index = 0; // index of the next promise to process
    const results = []; // store results in order
    let activeCount = 0; // number of in-flight promises

    // Execute the next promise
    const executeNext = () => {
      // All promises done — resolve
      if (index === promises.length && activeCount === 0) {
        resolve(results);
        return;
      }

      // While there's a free slot, run more promises
      if (activeCount < concurrency && index < promises.length) {
        const currentIndex = index++; // save the current promise's index
        activeCount++; // increment in-flight count

        promises[currentIndex]
          .then((result) => {
            results[currentIndex] = result; // save result by index
          })
          .catch((error) => {
            results[currentIndex] = error; // capture errors too
          })
          .finally(() => {
            activeCount--; // decrement in-flight count
            executeNext(); // run next
          });

        // Recursive call to keep filling slots
        executeNext();
      }
    };

    // Kick off
    executeNext();
  });
}



const promise1 = new Promise(resolve => setTimeout(() => resolve('Result 1'), 1000));
const promise2 = new Promise(resolve => setTimeout(() => resolve('Result 2'), 500));
const promise3 = new Promise(resolve => setTimeout(() => resolve('Result 3'), 300));
const promise4 = new Promise(resolve => setTimeout(() => resolve('Result 4'), 200));
const promise5 = new Promise(resolve => setTimeout(() => resolve('Result 5'), 400));

const promises = [promise1, promise2, promise3, promise4, promise5];

promisePool(promises, 2)
  .then(results => {
    console.log(results);
    // Results in order: ['Result 1', 'Result 2', 'Result 3', 'Result 4', 'Result 5']
  })
  .catch(err => console.error(err));

```
