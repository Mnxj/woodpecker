Prototype chain inheritance

> A child object can access properties and methods on the parent's prototype chain.
>
> **Modifying the prototype affects all child instances.**

```js
function Parent (){
    this.name = 'Parent';
}
Parent.prototype.sayHello = function(){
    console.log('hello')
}
function Child () {
    this.name = 'Child';
}
Child.prototype = new Parent();
let child = new Child();
child.sayHello();
```

Constructor inheritance

> Call the parent constructor inside the child constructor — only inherits instance properties.
>
>
>
> **Cannot** access properties/methods on the parent's prototype chain.

```js
function Parent (name){
    this.name = name;
}
function Child (name) {
    Parent.call(this, name)
}
let child = new Child('Child');
```

Combined inheritance

> Inherits both the parent constructor's properties and the methods on the parent's prototype.
>
> **Calls the parent constructor twice.**

```js
function Parent (name){
    this.name = name;
}
Parent.prototype.sayHello = function(){
    console.log('hello')
}
function Child (name) {
    Parent.call(this, name)
}
Child.prototype = new Parent('parent');
let child = new Child('Child');
child.sayHello();
console.log(child.name)
```

Prototypal inheritance

> Create a new object whose prototype points to an existing object.
>
> **Can inherit properties and methods but can't pass constructor arguments.**

```js
var Parent = {
    name: 'Parent',
    sayHello: function() {
        console.log('Hello')
    }
}
var child = Object.create(parent);
console.log(child.name)
child.sayHello();
```

Parasitic inheritance

> Enhances an object on top of prototypal inheritance.
>
> Hard to reuse.

```js
function createAnother(obj) {
    // Create a new object
    let clone = Object.create(obj);
    // Enhance it
    clone.sayHi = function() {
        console.log('Hi');
    };
    // Return the enhanced object
    return clone;
}

let person = {
    name: 'John',
    age: 30
};

let anotherPerson = createAnother(person);

console.log(anotherPerson.name);
anotherPerson.sayHi();
```

**Parasitic combination inheritance**

> Calls the parent constructor only once.

```js
// General-purpose version
function Parent(name) {
    this.name = name;
}
Parent.prototype.getName = function() {
    console.log(this.name);
};
function Child(name, age) {
    // Call parent constructor
    Parent.call(this, name);
    this.age = age;
}
function createObj(o) {
    // Inherit parent's prototype properties/methods without calling new Parent()
    function F() {}
    F.prototype = o;
    // Return an empty object whose prototype is parent's prototype
    return new F();
}

// Equivalent to: Child.prototype = Object.create(Parent.prototype)
Child.prototype = createObj(Parent.prototype);
Child.prototype.constructor = Child;

let child = new Child("tom", 12);
child.getName(); // tom
```

class

```js
class Parent {
    constructor(age) {
        this.age = age;
    }
    getName() {
        console.log(this.name);
    }
}
class Child extends Parent {
    constructor(name, age) {
        super(age);
        this.name = name;
    }
}
```

Mixin inheritance

> Merge methods from multiple objects into one object, then mix it into other objects for inheritance and code reuse.

### Hand-written class

```js
function selfClass(Child, Parent) {
    // Object.create's second argument defines properties with property/accessor descriptors
    Child.prototype = Object.create(Parent.prototype, {
        // Child inherits parent's prototype properties and methods
        constructor: {
            enumerable: false,
            configurable: false,
            writable: true,
            value: Child
        }
    });
    // Inherit parent's static properties and methods
    Object.setPrototypeOf(Child, Parent);
}

// Test
function Child() {
    this.name = 123;
}
function Parent() {}
// Define parent's static method getInfo
Parent.getInfo = function() {
    console.log("info");
};
Parent.prototype.getName = function() {
    console.log(this.name);
};
selfClass(Child, Parent);
Child.getInfo(); // info
let tom = new Child();
tom.getName(); // 123
```
