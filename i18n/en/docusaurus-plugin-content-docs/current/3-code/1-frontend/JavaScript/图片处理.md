### Image preloading

> Takes an image URL, creates a new `Image` object, and sets its `src`. When the image finishes loading, the `onload` event fires — you can log a message or do something else like adding the image to the page. If loading fails, `onerror` fires.

```js
function preloadImage(url) {
    // Create a new Image
    var img = new Image();
    // Set the src to the URL to preload
    img.src = url;
    // Optional listener for load completion
    img.onload = function() {
        console.log('Image loaded');
        // Do something — e.g. add to DOM
    };
    // Optional listener for load failure
    img.onerror = function() {
        console.error('Image failed to load');
    };
}

// Usage
preloadImage('path/to/your/image.jpg');
```




### Image lazy loading

```js
// HTML
// <img src="./loading.jpg" data-src="https://cube.elemecdn.com/6/94/4d3ea53c084bad6931a56d5158a48jpeg.jpeg">
// <img src="./loading.jpg" data-src="https://fuss10.elemecdn.com/e/5d/4a731a90594a4af544c0c25941171jpeg.jpeg">

function observerImg() {
  let imgList = document.getElementsByTagName("img");
  let observer = new IntersectionObserver(list => {
    // Callback receives an array
    list.forEach(item => {
      // Check whether the element is in the viewport
      if (item.intersectionRatio > 0) {
        // Set the img src
        item.target.src = item.target.getAttribute("data-src");
        // After setting src, stop observing
        observer.unobserve(item.target);
      }
    });
  });
  for (let i = 0; i < imgList.length; i++) {
    // Observe each img
    observer.observe(imgList[i]);
  }
}


function lazyLoadImages() {
  const imgs = document.querySelectorAll('img[data-src]');

  imgs.forEach(img => {
    const rect = img.getBoundingClientRect();
    if (rect.top < window.innerHeight && rect.bottom > 0) {
      img.src = img.getAttribute('data-src');
      img.removeAttribute('data-src');
    }
  });
}

// Call at appropriate times, e.g. on scroll
window.addEventListener('scroll', lazyLoadImages);
```



### Get all image srcs on a page (be aware that CSS backgrounds need a special DOM approach)

1. Get all nodes in the document with `document.getElementsByTagName('*')`.

2. Iterate nodes.

3. Check whether the node is an image: for each node, check `node.nodeType === 1` (element node) and whether `tagName === 'IMG'`.

4. Get the image's `src` if it's an image node, via `node.getAttribute('src')`.

5. Check CSS background images: for each node, check whether the computed style has a background image. Use `window.getComputedStyle(node)` and inspect `backgroundImage` for a URL.

6. Extract the URL from `backgroundImage` if present.

```js
function getAllImageSrcs() {
  const images = [];
  const nodes = document.getElementsByTagName('*');

  for (let i = 0; i < nodes.length; i++) {
    const node = nodes[i];

    if (node.nodeType === 1 && node.tagName === 'IMG') {
      images.push(node.getAttribute('src'));
    } else if (window.getComputedStyle(node).backgroundImage.includes('url')) {
      const backgroundImage = window.getComputedStyle(node).backgroundImage.replace(/^url\(['"]?/, '').replace(/['"]?\)$/, '');
      images.push(backgroundImage);
    }
  }

  return images;
}

console.log(getAllImageSrcs());
```
