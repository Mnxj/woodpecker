### Convert an array of objects into a tree

```js
/**
* Convert a list into a tree structure
* Given list data with parent/child levels, convert into objects and finally a tree,
* recording each node's chain of parent codes in order.
* Write a function that performs this transformation on demoList and any data of the same shape.
* demoList => demoListTree. Also state the time complexity.
*/

// Shallow copy of objects — modifying a reference modifies the object
const demoList = [
  { code: 1, pcode: null, level: 1 },
  { code: 2, pcode: 1, level: 2 },
  { code: 3, pcode: 2, level: 3 },
  { code: 4, pcode: null, level: 1 },
  { code: 5, pcode: 4, level: 2 },
  { code: 6, pcode: null, level: 1 }
]

function listToTree(arr) {
  const map = {};
  const tree = [];
  arr.forEach(element => {
    element.pcodes = []
    map[element.code] = element;
  });
  arr.forEach(element => {
    const pitem = map[element.pcode];
    if(pitem) {
      if(!pitem.children) {
        pitem.children = [];
      }
      element.pcodes = [...pitem.pcodes.concat(pitem.code)];
      pitem.children.push(element)

    } else {
      tree.push(element);
    }

  });

  return tree;
};
```

### Flatten a nested object so each key maps to a primitive value

```js
function flattenObject(obj) {
  const result = {};

  for (const key in obj) {
    if (obj.hasOwnProperty(key)) {
      if (typeof obj[key] === 'object' && obj[key] !== null && !Array.isArray(obj[key])) {
        const flattenedSubObj = flattenObject(obj[key]);
        for (const subKey in flattenedSubObj) {
          result[key + '.' + subKey] = flattenedSubObj[subKey];
        }
      } else {
        result[key] = obj[key];
      }
    }
  }

  return result;
}

const nestedObject = {
  a: 1,
  b: {
    c: 2,
    d: {
      e: 3
    }
  }
};

console.log(flattenObject(nestedObject));
```

### Write a `diff` function (deep compare)

```js
function deepEqual(a, b) {
    // Primitives equality
    if (a === b) return true;

    // null or undefined
    if (a == null || b == null) return false;

    // Object or array?
    if (typeof a !== 'object' || typeof b !== 'object') return false;

    // Compare lengths
    if (Array.isArray(a) && Array.isArray(b)) {
        if (a.length !== b.length) return false;
    } else {
        // Number of keys
        if (Object.keys(a).length !== Object.keys(b).length) return false;
    }

    // Recursively compare each property/element
    for (let key in a) {
        if (!b.hasOwnProperty(key) || !deepEqual(a[key], b[key])) {
            return false;
        }
    }

    return true;
}

// Usage
const obj1 = { a: 1, b: { c: 2 } };
const obj2 = { a: 1, b: { c: 2 } };
const obj3 = { a: 1, b: { c: 3 } };

console.log(deepEqual(obj1, obj2)); // true
console.log(deepEqual(obj1, obj3)); // false
```

### Convert nested object keys from snake_case to camelCase

```js
function convertKeysToCamelCase(obj) {
    if (Array.isArray(obj)) {
        // Recursively handle each element if array
        return obj.map(item => convertKeysToCamelCase(item));
    }
    if (obj && typeof obj === 'object') {
        // Recursively handle each property if object
        const result = {};
        Object.keys(obj).forEach(item => {
            const key = item.replace(/(_\w)/g, (match) => match[1].toUpperCase());
            result[key] = convertKeysToCamelCase(obj[item])
        });
        return result;
    }
    // Otherwise return original value
    return obj;
}

// Usage
const nestedObject = {
    first_name: 'John',
    last_name: 'Doe',
    address: {
        street_address: '123 Main St',
        city: 'Anytown',
        zip_code: '12345'
    }
};

const camelCaseObject = convertKeysToCamelCase(nestedObject);
console.log(camelCaseObject);
```

### Express the following ES class with ES5

```js
class person { constructor (name) { this.name = name; } call () { return "I'm" + this.name; } static isPerson() { return true; } }
```

```js
// Use a constructor to simulate a class
function Person(name) {
    // `this` inside the constructor refers to the new instance
    this.name = name;
}

// Add methods to Person's prototype
Person.prototype.call = function() {
    return "I'm " + this.name;
};

// Add a static method to Person
Person.isPerson = function() {
    return true;
};

// Usage
var person1 = new Person('Alice');
console.log(person1.call()); // I'm Alice

console.log(Person.isPerson()); // true
```

### Implement a `commafy` function that adds commas every three digits in the integer part, e.g. 12000000.11 → 12,000,000.11.

```js
function commafy(number) {
  // Use en-US locale to format numbers — guarantees comma as the thousands separator
  return number.toLocaleString('en-US');
}
// Usage
const num = -12000000.11;
```
