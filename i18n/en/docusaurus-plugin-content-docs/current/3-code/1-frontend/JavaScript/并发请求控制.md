### How to control concurrent request count?

```js
const request = (queue, max) => {
    return new Promise((resolve, reject) => {
        function execute(q, len) {
            if(!q.length) resolve();
            while(q.length && len > 0) {
                len--;
                const task = q.shift();
                Promise.resolve(task())
                    .then(res => resolve(res))
                    .catch(err => reject(err))
                    .finally(() => {
                        execute(q, len++)
                    })
            }
        }
        execute(queue, max)
    })
}
```

How to do serial requests with up to 3 retries on error?

```js
function serialRequest(promises) {
    return new Promise((resolve, reject) => {
        let index = 0;
        function nextRequest() {
            if (index >= promises.length) resolve()
            const task = promises.shift();
            let retryCount = 0;
            function attemptRequest() {
                Promise.resolve(task())
                    .then(data => {
                        index++; // move to next request
                        return nextRequest(); // continue
                    })
                    .catch(error => {
                        retryCount++;
                        if (retryCount < 3) {
                            return attemptRequest(); // retry
                        } else {
                            index++; // move to next even on failure
                            return nextRequest(); // continue
                        }
                    });
            }
            return attemptRequest();
        }
        return nextRequest();
    })
}
```

Implement a JS retry helper for failed requests

```js
function reTry(asyncFunc, times) {
    return new Promise(async (resolve, reject) => {
        function reTryFunc(times) {
            Promise.resolve(asyncFunc())
                .then((res) => resolve(res))
                .catch((err) => {
                    if (times > 0) {
                        reTryFunc(times - 1)
                    } else {
                        reject(err)
                    }
                })
        }
        reTryFunc(times)
    })
}
```
