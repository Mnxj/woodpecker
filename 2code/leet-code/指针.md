### 二进制求和

```js
var addBinary = function(a, b) {
    let i = a.length - 1       // a 的最大索引
    let j = b.length - 1       // b 的最大索引
    let res = ''
    let c = 0                        // 记录进位
    while( i >= 0 || j >=0 || c ){   // 从后面往前加
      let aa = +a[ i-- ] || 0
      let bb = +b[ j-- ] || 0
      let total = aa + bb + c
      c = total / 2 | 0
      total %= 2
      res = total + res;
    }
    return res
};
```
